cmake_minimum_required(VERSION 2.6)

project( 01_window )

find_package(GLFW3 3.2 REQUIRED)

if (GLFW_FOUND)

  set( SOURCES
    src/01_window.cpp
  )

  set ( HEADERS
  )

  include_directories( "."
                       "inc"
                       "$ENV{VULKAN_SDK}/Samples/API-Samples/utils"
                       ${GLFW_INCLUDE_DIRS}
                     )

  add_definitions(-DNOMINMAX)

  add_executable( 01_window
    ${SOURCES}
    ${HEADERS}
  )

  target_link_libraries( 01_window
    ${GLFW_LIBRARIES}
    VkHLF
  )

  set_target_properties( 01_window PROPERTIES FOLDER "Samples")
else()
  message("GLUT not found, disabling GLUTHelloVulkan.")
endif()
